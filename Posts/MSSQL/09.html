public void SELECT()
        {
            SqlCommand cmd = new SqlCommand("SELECT * FROM table1", con);
            try
            {
                dr = cmd.ExecuteReader();
 
                while (dr.Read())
                {
                    Console.Write(String.Format(" {0}, {1}", dr[0], dr[1]));
                    Console.Write("\n");
 
                }
                Console.WriteLine("SELECT 정상적 작동");
            }
            catch (Exception e)
            {
                Console.WriteLine("SELECT 오류");
                Console.WriteLine(e.Message);
            }
