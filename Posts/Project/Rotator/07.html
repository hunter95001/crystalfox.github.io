        private void sort(int[] sortnum)
        {
            int swap;                                                    // 임시변수
            for (int i = 0; i < sortnum.GetLength(0) - 1; i++)
                for (int j = i + 1; j < sortnum.GetLength(0); j++)
                    if (sortnum[i] > sortnum[j])
                    {
                        swap = sortnum[i];
                        sortnum[i] = sortnum[j];
                        sortnum[j] = swap;
                    }
        }
