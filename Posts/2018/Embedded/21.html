#define LM35_PIN A0
 
#define V_OFFESET 0
#define SECALE_FACTOR 10
double celsius, fahrenheit;
 
 
void setup() {
  // put your setup code here, to run once:
  Serial.begin(115200);
  Serial.println("Start...");
}
 
void loop() {
  // put your main code here, to run repeatedly:
  long volt =analogRead(LM35_PIN); //아날로그값 읽음
  volt  = volt * 5000 /1024; //mV로 바꾸는 작업
  Serial.println(volt);  
  celsius = (volt -V_OFFESET) /SECALE_FACTOR; //섭시 온도
  fahrenheit = (celsius*1.8)+32; //화씨온도
  Serial.println("Temp. :");  
  Serial.print(celsius,2);
  Serial.print("  C /");  
  Serial.print(fahrenheit,2);
  Serial.println("  F ");  
  Serial.println("==================================="); 
 
  delay(2000);
}
