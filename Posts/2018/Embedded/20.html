#include <LiquidCrystal_I2C.h>
// DS1302_Serial_Easy (C)2010 Henning Karlsen
// web: http://www.henningkarlsen.com/electronics
//
// Adopted for DS1302RTC library by Timur Maksimov 2014
//
// A quick demo of how to use my DS1302-library to
// quickly send time and date information over a serial link
//
// I assume you know how to connect the DS1302.
// DS1302: CE pin -> Arduino Digital 27
// I/O pin -> Arduino Digital 29
// SCLK pin -> Arduino Digital 31
// VCC pin -> Arduino Digital 33
// GND pin -> Arduino Digital 35
#include <TimeLib.h>
#include <DS1302RTC.h>
LiquidCrystal_I2C lcd(0x3F,16,2);
// Set pins: CE, IO,CLK
DS1302RTC RTC(2, 3, 4);
// Optional connection for RTC module
#define DS1302_GND_PIN 33
#define DS1302_VCC_PIN 35
int cdspin =A4;
int Year,Month,Day,Wday,Hour,Minute,Second,cds=0;
 
void setup()
{
// Setup Serial connection
Serial.begin(115200);
 
Serial.println("DS1302RTC Read Test");
Serial.println("-------------------");
 
// Activate RTC module
digitalWrite(DS1302_GND_PIN, LOW);
pinMode(DS1302_GND_PIN, OUTPUT);
digitalWrite(DS1302_VCC_PIN, HIGH);
pinMode(DS1302_VCC_PIN, OUTPUT);
 
Serial.println("RTC module activated");
Serial.println();
delay(500);
 
if (RTC.haltRTC()) {
Serial.println("The DS1302 is stopped. Please run the SetTime");
Serial.println("example to initialize the time and begin running.");
Serial.println();
}
if (!RTC.writeEN()) {
Serial.println("The DS1302 is write protected. This normal.");
Serial.println();
}
lcd.begin();
lcd.backlight();
lcd.print("RTC Clock");
 
delay(5000);
}
void loop()

tmElements_t tm;
char buf[40];
analogRead(cdspin);

if((Year !=tmYearToCalendar(tm.Year)) || (Month!=tm.Month) || (Day !=tm.Day) || (Wday !=tm.Wday)) {
 
Year=tmYearToCalendar(tm.Year);
Month=tm.Month;
Day=tm.Day;
 
 
lcd.setCursor(0,0);
sprintf(buf, "%4d:%2d:%2d ",tmYearToCalendar(tm.Year),tm.Month,tm.Day);
lcd.print(buf);
week(tm.Wday);
}

void print2digits(int number) {
if (number >= 0 && number < 10)
Serial.write('0');
Serial.print(number);
}
 
void week(int number) {
switch(number){
case 1:
lcd.print("SUN");
break;
case 2:
lcd.print("MON");
break;
case 3:
lcd.print("TUE");
break;
case 4:
lcd.print("WED");
break;
case 5:
lcd.print("THU");
break;
case 6:
lcd.print("FRI");
break;
case 7:
lcd.print("SAT");
break;
 
}
}
