#define TRIG_PIN  9
#define ECHO_PIN  10
#define SOUND_SPEED 0.034 //cm/usec
 
const int IN0=2;
const int IN1=3;
const int IN2=4;
const int IN3=5;
int step=0;
int count=0;
//bool count =true;
 
 
 
void stepPulse(int step){
  switch(step){
  case 0:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
case 1:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, HIGH);
break;
case 2:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 3:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 4:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 5:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 6:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 7:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
default:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
    
  }
  delay(1);
}
void setup()
{
pinMode(IN0, OUTPUT);
pinMode(IN1, OUTPUT);
pinMode(IN2, OUTPUT);
pinMode(IN3, OUTPUT);
 pinMode(TRIG_PIN,OUTPUT);
 pinMode(ECHO_PIN,INPUT);
 
 Serial.begin(115200);
 Serial.println("Start...");
}
 
int countB =0;// 세마포어 
int ultra_wave(){
  digitalWrite(TRIG_PIN,LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN,HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN,LOW);
  int duration = pulseIn(ECHO_PIN,HIGH);
  int distance = duration * SOUND_SPEED / 2;
  return distance;
  }
 
void Motor(){
  for(int i=0;i<64*16;i++){
  stepPulse(step);
  if(step++>7)//회전
    step=0;
    }
    }
void tooggled_Motor(){
  
  for(int i=0;i<64*16;i++){
  stepPulse(step);
  if(step--<0)//역회전
    step=7;
  }
  }
int dist;
void loop()
{
  dist =ultra_wave();
  Serial.print("dist : ");
  Serial.println(dist);
  Serial.print("count : ");
  Serial.println(count);
 
if(dist <30 &&count==0 ){ //차량이 오면 인식해서 문을 열어주는 상태 
  Motor(); 
  count =1; 
  } 
else if(dist <30 &&count==1 ){ //차량이 계속 있으면 문을 열어주는 행동을 유지하는 상태 
  delay(1000); 
  } 
else if(dist >30 &&count==1 ){ //차량이 지나가면 문을 닫아 주는 상태 
   dist =ultra_wave();
delay(1000); 
countB++; 
if(countB>2){ 
 tooggled_Motor(); 
count=0; 
} 
  } 
  else if(dist >30 &&count==0 ){ //아무것도 인식하지 않는 상태 
     dist =ultra_wave();
 delay(1000); 
  }
  
  
 
 
}
