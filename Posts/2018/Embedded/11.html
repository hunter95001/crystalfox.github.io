#include <Bounce2.h>
 
const int fndStartPin =24;
const int fndEndPin =31;
const int btnPin =2;
 
void fndWrite(int num){
  //FND segment code table for comm-cathode FND
  static unsigned char fndSegCode[] ={
    0x3f,0x06,0x5b,0x4f,0x66,0x6d,
    0x7d,0x07,0x7f,0x6f,0x77,0x7c,
    0x39,0x5e,0x79,0x71
   
    };
    unsigned char outcode;
    int value;
    if(num<0) num=0;
    else if(num>=16)num=15;
    outcode =fndSegCode[num];
 
    for(int i=fndStartPin; i<=fndEndPin; i++){
      value =(outcode & 0x01) ? LOW :HIGH;
      digitalWrite(i,value);
      outcode >>=1;
      }
  }
 
void setup() {
  // put your setup code here, to run once:
 for(int i=fndStartPin; i<=fndEndPin; i++)
 pinMode(i,OUTPUT);
 
 pinMode(btnPin,INPUT);
}
 
void loop() {
  // put your main code here, to run repeatedly:
for(int i=0; i<16; i++){
fndWrite(i);
delay(1000);
}
}
