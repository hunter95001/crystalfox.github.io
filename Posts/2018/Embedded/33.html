const int IN0=22;

const int IN3=25;

void stepPulse(int step){
switch(step){
case 0:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
case 1:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, HIGH);
break;
case 2:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 3:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 4:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 5:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 6:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 7:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
default:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
 
}
delay(1);
}
 
void Motor(){
for(int i=0;i<64*16;i++){
stepPulse(step);
if(step++>7)//회전
step=0;
}
}
 
void tooggled_Motor(){
 
for(int i=0;i<64*16;i++){
stepPulse(step);
if(step--<0)//역회전
step=7;
}
}

void Music(){ //엘리스를 위하여 ~ 라이브 러리

void sing(int eum, int isLong) {
long lele = 125000;
 
if(isLong == 3) lele = 375000;
else if(isLong == 6) lele = 750000;
 
for(long i=0; i< lele ; i+= eum) {
digitalWrite(PIEZO, HIGH);
delayMicroseconds(eum/2);
digitalWrite(PIEZO, LOW);
delayMicroseconds(eum/2);
}
}
 
void stepPulse(int step){
switch(step){
case 0:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
case 1:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, HIGH);
break;
case 2:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 3:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, HIGH);
digitalWrite(IN3, LOW);
break;
case 4:
digitalWrite(IN0, LOW);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 5:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, HIGH);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 6:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
case 7:
digitalWrite(IN0, HIGH);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, HIGH);
break;
default:
digitalWrite(IN0, LOW);
digitalWrite(IN1, LOW);
digitalWrite(IN2, LOW);
digitalWrite(IN3, LOW);
break;
 
}
delay(1);
}
 
bool sema =true; //정방향 회전을 1번만 해주기 위해서 세마포어를 걸어줌
bool semaB =false; //역방향 회전을 1번만 해주기 위해서 세마포어를 걸어줌
void Motor(){
if(sema ==true){
for(int i=0;i<64*16;i++){
stepPulse(step);
if(step++>7)//회전
step=0;
}
sema =false;
semaB=true;
}
}
 
void tooggled_Motor(){
if(semaB==true){
for(int i=0;i<64*16;i++){
stepPulse(step);
if(step--<0)//역회전
step=7;
}
semaB=false;
sema=true;
}
}

void initCard(){
// Look for new cards
if ( ! mfrc522.PICC_IsNewCardPresent()) {
return;
}
// Select one of the cards
if ( ! mfrc522.PICC_ReadCardSerial()) {
return;
}
// Dump debug info about the card; PICC_HaltA() is automatically called
mfrc522.PICC_DumpToSerial(&(mfrc522.uid));
}

void login(){
boolean loops =true;
while(loops){
digitalWrite(TRIG_PIN,LOW);
delayMicroseconds(2);
digitalWrite(TRIG_PIN,HIGH);
delayMicroseconds(10);
digitalWrite(TRIG_PIN,LOW);
int duration = pulseIn(ECHO_PIN,HIGH);
int distance = duration * SOUND_SPEED / 2;
Motor(); //정방향 회전
Music(); //엘리스를 위하여
Blink(); //문이 열리면 반짝 반짝
setColor(255, 0); // red
 
lcd.clear();
lcd.setCursor(0,16);
lcd.print("Open"); //문이 열립니다
 
if(mfrc522.uid.uidByte[0]==KeyCard[0]){
lcd.setCursor(0,0);
lcd.print("KeyCard Login"); //키 카드로 문이 열립니다
}
 
if(mfrc522.uid.uidByte[0]==WhiteCard[0]){
lcd.setCursor(0,0);
lcd.print("White Card Login"); //화이트 카드로 문이 열립니다
}
if((prePassword[3]==password[3])){
lcd.setCursor(0,0);
lcd.print("Password"); //비밀번호 입력
lcd.print(password[0]); //비밀번호 입력
lcd.print(password[1]); //비밀번호 입력
lcd.print(password[2]); //비밀번호 입력
lcd.print(password[3]); //비밀번호 입력
}
timecount++; //34초 시간 변수
if(timecount>34 && distance>20){ // 34초가 지나면 문을 닫습니다. 엘리스를 위하여 노래가 잘 들리기 때문입니다 가이 있으면 문이 닫히지 않습니다
mfrc522.uid.uidByte[0]=0; //이것을 바꿔주면 RFID의 인식한 사용자를 바꿔줍니다
timecount=0;
for(int i=0; i<4; i++){
prePassword[i]='K';
precount=0;
}
loops=false;
}
}
}

void setup() {
pinMode(redPin, OUTPUT);
pinMode(greenPin, OUTPUT);
pinMode(PIEZO, OUTPUT);
pinMode(IN0, OUTPUT);
pinMode(IN1, OUTPUT);
pinMode(IN2, OUTPUT);
pinMode(IN3, OUTPUT);
pinMode(TRIG_PIN,OUTPUT);
pinMode(ECHO_PIN,INPUT);
Serial.begin(115200); //PC와의 속도
SerialBLE.begin(38400); //블루투스와의 연결
Blynk.begin(SerialBLE, auth);
Serial.println("Waiting for connections...");
while (!Serial); // Do nothing if no serial port is opened (added for Arduinos based on ATMEGA32U4)
SPI.begin(); // Init SPI bus
mfrc522.PCD_Init(); // Init MFRC522
mfrc522.PCD_DumpVersionToSerial(); // Show details of PCD - MFRC522 Card Reader details
Serial.println(F("Scan PICC to see UID, SAK, type, and data blocks..."));
lcd.begin();
lcd.backlight();
lcd.print("Start..");
}
