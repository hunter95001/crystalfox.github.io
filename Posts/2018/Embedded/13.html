#include <Bounce2.h>
 
const int clockPin = 2; 
const int dataPin = 3; 
const int latchPin = 4;
const int btnPin =10;
const int btn1Pin =8;
const int pin1 =53;
const int pin2 =52;
const int pin3 =51;
const int pin4 =50;
int lastBtnState =HIGH;
int btnInputCnt=0;
Bounce debouncer =Bounce();
 
 
 
void change(int num){
  if(num ==0){
digitalWrite(pin1, LOW);
digitalWrite(pin2, HIGH);
digitalWrite(pin3, HIGH);
digitalWrite(pin4, HIGH);
  }
  
  if(num ==1){
digitalWrite(pin1, HIGH);
digitalWrite(pin2, LOW);
digitalWrite(pin3, HIGH);
digitalWrite(pin4, HIGH);
  }
 
  if(num ==2){
digitalWrite(pin1, HIGH);
digitalWrite(pin2, HIGH);
digitalWrite(pin3, LOW);
digitalWrite(pin4, HIGH);
  }
 
    if(num ==3){
digitalWrite(pin1, HIGH);
digitalWrite(pin2, HIGH);
digitalWrite(pin3, HIGH);
digitalWrite(pin4, LOW);
  }
  
  }
  
boolean chekRisingEdge(){
  boolean edge;
  int state;
  
  debouncer.update();
  state =debouncer.read();
  
  edge =(state&& (state !=lastBtnState))? true: false;
  lastBtnState =state;
  
  return edge;
  }
 
 
void fndWriteBySR(int num) {
static unsigned char fnd_segcode[] = {
0x3F, 0x06, 0x5B, 0x4F, 
0x66, 0x6D, 0x7D, 0x07,
0x7F, 0x6F};
 
unsigned char outcode;
outcode=fnd_segcode[num];
if (num < 0) num = 0;
else if (num >= 10) num = 9;
digitalWrite(latchPin, LOW); // inactive latch signal 
shiftOut(dataPin, clockPin, MSBFIRST, outcode); 
digitalWrite(latchPin, HIGH); // active latch signal
}
 
void setup() {
Serial.begin(9600);
pinMode(dataPin, OUTPUT);
pinMode(clockPin, OUTPUT); 
pinMode(latchPin, OUTPUT);
pinMode(btnPin, INPUT_PULLUP);
pinMode(btn1Pin, INPUT_PULLUP);
 
pinMode(pin1,OUTPUT);
pinMode(pin2,OUTPUT);
pinMode(pin3,OUTPUT);
pinMode(pin4,OUTPUT);
 
digitalWrite(pin1, LOW);
digitalWrite(pin2, LOW);
digitalWrite(pin3, LOW);
digitalWrite(pin4, LOW);
digitalWrite(latchPin, LOW);
}
int num100=0;
int num10=0;
int num1=0;
int num=0;
void reset(){
    num100=0;
    num10=0;
    num1=0;
    num=0;
  }
void Stop(int i,int num, int num1, int num10, int num100){
   if(i==0){
  change(0);
   fndWriteBySR(num);
   delay(1);
    }
   if(i==1){
  change(1);
   fndWriteBySR(num1);
   delay(1);
    }
   if(i==2){
  change(2);
   fndWriteBySR(num10);
   delay(1);
    }
   if(i==3){
  change(3);
   fndWriteBySR(num100);
   delay(1);
    }
}
 
void loop() {
 
for( num100=0; num100<10; num100++) //1000에 단위
for(num10=0; num10<10; num10++) //100에 단위
for(num1=0; num1<10; num1++) //10에 단위
for (num=0; num<10; num++) { //1에 단위
  for(int j=0; j<55; j++){
  for(int i=0; i<4; i++){
    
  
 
   
      while(digitalRead(btnPin)==LOW){
         if(btnInputCnt==0)
        for(int m=0; m<4; m++)
  Stop(m,num,num1,num10,num100);
   if(btnInputCnt==1)
   break;
    }  
 
    if (digitalRead(btn1Pin) ==LOW){
  reset();
    }
  
 
  
  if(i==0){
  change(0);
   fndWriteBySR(num);
   delay(1);
    }
   if(i==1){
  change(1);
   fndWriteBySR(num1);
   delay(1);
    }
   if(i==2){
  change(2);
   fndWriteBySR(num10);
   delay(1);
    }
   if(i==3){
  change(3);
   fndWriteBySR(num100);
   delay(1);
    }
 
  }
  }
 
}
