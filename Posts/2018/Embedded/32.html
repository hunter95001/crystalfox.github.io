Bluetooth는 게이트웨이 역할.
IOT Gateware (Bluetooth)
기존의 Bluetooth를 이용한 하드웨어 제품이 많이 있다. 이것을 인터넷이랑 연결 시킴으로써 사용한다.



우리 보드는 HC06으로 Slave모드 밖에 없기 때문에 Key,State를 사용하지 않아서 핀 4개.
Commend Mode - Key - High , 전원 공급



저항을 사용 하면 안된다!
높은 전압에서는 저항이 열받아서 터져버린다.


예제 - 4번 - MultiSerial

AT+VERSION
AT+BAUD6 = (4=>9600) (6=>38400)
->전송속도를 38400을로 바꿔주셔야 합니다.
AT+NAMESILLA21
->블루투스 이름을 SILLA21바꿔줍니다
AT+PIN4321
->비밀번호 4321

#define BLYNK_PRINT Serial
#include <BlynkSimpleSerialBLE.h>
 
char auth[] = "f939b20f854843ccbb3f707cb57be153";
 
//SoftwareSerial SerialBLE(10, 11); // RX, TX <-블루투스 를 연결합니다
#define SerialBLE Serial1 //메크로 선언 -> SerialBLE 가 실행될떄 SerialSerial1이 됩니다
// This function will be called every time Slider Widget
// in Blynk app writes values to the Virtual Pin 1
BLYNK_WRITE(V1) //메크로 V1= 가상핀 
{
  int pinValue = param.asInt(); // assigning incoming value from pin V1 to a variable
 
 
  digitalWrite(2,pinValue);
  
  Serial.print("V1 Button value is: ");
  Serial.println(pinValue);
}
 
BLYNK_WRITE(V2) //메크로 V1= 가상핀 
{
  int pinValue = param.asInt(); // assigning incoming value from pin V1 to a variable
 
 
  analogWrite(2,map(pinValue,0,100,0,255));
  
  Serial.print("V2 Sillder value is: ");
  Serial.println(pinValue);
}
 
void setup()
{
  // Debug console
  Serial.begin(115200); //PC와의 속도
 
  SerialBLE.begin(38400); //블루투스와의 연결
  Blynk.begin(SerialBLE, auth);
  pinMode(2,OUTPUT);//LED 2번
 
  Serial.println("Waiting for connections...");
}
 
void loop()
{
  Blynk.run();
}
