#include <LiquidCrystal_I2C.h>
// DS1302_Serial_Easy (C)2010 Henning Karlsen
// web: http://www.henningkarlsen.com/electronics
//
// Adopted for DS1302RTC library by Timur Maksimov 2014
//
// A quick demo of how to use my DS1302-library to 
// quickly send time and date information over a serial link
//
// I assume you know how to connect the DS1302.
// DS1302: CE pin -> Arduino Digital 27
// I/O pin -> Arduino Digital 29
// SCLK pin -> Arduino Digital 31
// VCC pin -> Arduino Digital 33
// GND pin -> Arduino Digital 35
#include <TimeLib.h>
#include <DS1302RTC.h>
#include "DHT.h"
#define DHTTYPE DHT11
#define DHTPIN 12
LiquidCrystal_I2C lcd(0x3F,16,2);
// Set pins: CE, IO,CLK
DS1302RTC RTC(2, 3, 4);
int Year,Month,Day,Wday,Hour,Minute,Second,cds=0;
DHT dht(DHTPIN, DHTTYPE);
int count =0;
int sw = 8; // 스위치(SW) 핀 설정
int state = LOW; // LED 상태
int previous = HIGH; // SW 이전 상태
long time = 0; // LED가 ON/OFF 토글된 마지막 시간
long debounce = 100; // Debounce 타임 설정
void setup()
{
// Setup Serial connection
Serial.begin(115200); 
Serial.println("DS1302RTC Read Test");
Serial.println("-------------------");
pinMode(sw, INPUT_PULLUP); // SW 를 설정, 아두이노 풀업저항 사용
Serial.println("RTC module activated");
Serial.println();
delay(500);
if (RTC.haltRTC()) {
Serial.println("The DS1302 is stopped. Please run the SetTime");
Serial.println("example to initialize the time and begin running.");
Serial.println();
}
if (!RTC.writeEN()) {
Serial.println("The DS1302 is write protected. This normal.");
Serial.println();
}
lcd.begin();
lcd.backlight();
lcd.print("RTC Clock");
Serial.println("DHTxx test!");
delay(5000);
}
void loop(){
tmElements_t tm;
char buf[40]; 
float h = dht.readHumidity();
float t = dht.readTemperature();
float f = dht.readTemperature(true);
if (isnan(h) || isnan(t) || isnan(f)) {
Serial.println("Failed to read from DHT sensor!");
return;
}
float hif = dht.computeHeatIndex(f, h);
// Compute heat index in Celsius (isFahreheit = false)
float hic = dht.computeHeatIndex(t, h, false);
if(digitalRead(sw)==LOW && millis() - time > debounce)
toggled();
if(previous ==0){
lcd.clear();
lcd.setCursor(0,0);
lcd.print("Humidity");
lcd.print(h);
lcd.setCursor(0,1);
lcd.print("temp.:");
lcd.print(t);
lcd.print("*C");
}
if(previous ==1)
if (! RTC.read(tm)) {
Year=tmYearToCalendar(tm.Year);
Month=tm.Month;
Day=tm.Day; 
Hour=tm.Hour;
Minute=tm.Minute;
Second=tm.Second;
lcd.clear();
lcd.setCursor(0,0);
sprintf(buf, "%4d:%2d:%2d ",tmYearToCalendar(tm.Year),tm.Month,tm.Day);
lcd.print(buf);
week(tm.Wday);
lcd.setCursor(0,16);
sprintf(buf, "%2d:%2d:%02d",tm.Hour,tm.Minute,tm.Second );
lcd.print(buf);
}
// Wait one second before repeating :)
delay (1000);
}
void week(int number) {
switch(number){
case 1:
lcd.print("SUN");
break;
case 2:
lcd.print("MON");
break;
case 3:
lcd.print("TUE");
break;
case 4:
lcd.print("WED");
break;
case 5:
lcd.print("THU");
break;
case 6:
lcd.print("FRI");
break;
case 7:
lcd.print("SAT");
break;
}
}
void toggled(){
previous =previous == HIGH?LOW:HIGH;
}
