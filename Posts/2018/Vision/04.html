using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
 
namespace WindowsFormsApp1
{
    public partial class Form1 : Form
    {
        Image image;
        public Form1()
        {
            InitializeComponent();
        }
 
        private void openToolStripMenuItem_Click(object sender, EventArgs e)
        {
            Graphics gr = CreateGraphics();
            string name = "All Files(*.*)|*.*|Bitmap File(*.bmp)|*.bmp|";
            name = name + "Gif File(*.gif)|*.gif|jpeg File(*.jpg)|*.jpg";
            openFileDialog1.Title = "타이틀";
            openFileDialog1.Filter = name;
            if (openFileDialog1.ShowDialog() == DialogResult.OK)
            {
                string strName = openFileDialog1.FileName;
 
                image = Image.FromFile(strName);
 
                pictureBox1.Image =image;
 
            }
 
            gr.Dispose();
        }
 
        private void grayToolStripMenuItem_Click(object sender, EventArgs e)
        {
            int brightness;
            Color color, gray;
            Graphics gr = CreateGraphics();
            Bitmap bitmap = new Bitmap(image);
            for (int x = 0; x < image.Width; x++)
                for (int y = 0; y < image.Height; y++)
                {
                    color = bitmap.GetPixel(x, y);
                    brightness = (int)(0.299 * color.R + 0.587 * color.G + 0.114 * color.B); //int형으로 변환해주는 이유는 소수점이하 버림을 할려고
                    gray = Color.FromArgb(brightness, brightness, brightness);
                    bitmap.SetPixel(x, y, gray);
                }
            pictureBox2.Image = bitmap;
            gr.Dispose();
 
        }
    }
}
