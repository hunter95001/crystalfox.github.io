CREATE OR REPLACE FUNCTION cal_total
    (v_kor student2018.hangle%TYPE,
    v_eng student2018.english%TYPE,
    v_math student2018.math%TYPE,
    v_toeic student2018.toeic%TYPE)
    RETURN NUMBER
IS
    v_avg NUMBER;
    v_total NUMBER;
BEGIN    
    v_avg := ROUND((v_kor+v_eng+v_math)/3*0.8,1);
    v_total := ROUND(v_avg+v_toeic*0.2);
    RETURN v_total ;
EXCEPTION
    WHEN  NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE(' 존재하지 않는 교수번호입니다.');
    WHEN  TOO_MANY_ROWS THEN
    DBMS_OUTPUT.PUT_LINE(' 값이 여러개 있습니다.');
END;

SELECT studno 학번,name 학생이름 ,grade 학년,hangle,english,math,TOEIC, cal_total(hangle, english, math, toeic) 총점
FROM student2018
ORDER BY 총점 DESC
