WITH e
AS(
        SELECT carno,owner, maker, capacity , type,
                    CASE  
                    WHEN capacity <= 1000 THEN TRUNC(capacity *18,-1)
                    WHEN capacity <= 1600 THEN TRUNC(capacity *18,-1)
                    WHEN capacity <= 2000 THEN TRUNC(capacity *19,-1)
                    WHEN capacity <= 3000 THEN TRUNC(capacity *19,-1)
                    ELSE TRUNC(capacity *24,-1)
                       END  tax,
                    CASE  
                    WHEN capacity <= 1000 THEN TRUNC(capacity *80,-1)
                    WHEN capacity <= 1600 THEN TRUNC(capacity *140,-1)
                    WHEN capacity <= 2000 THEN TRUNC(capacity *200,-1)
                    WHEN capacity <= 3000 THEN TRUNC(capacity *200,-1)
                    ELSE TRUNC(capacity *200,-1)
                       END  taxs
FROM vehicle2018)
SELECT e.carno 차량번호,e.owner 소유자,e.maker 제조회사,e.capacity 배기량,e.type 구분, DECODE(e.type,'비영업',e.taxs,
                                            e.tax) 세금
FROM e
