공과대 학 소속의
학과번호 학과이름 소속을 출력하시오.

SELECT d1.deptno "학과번호", d1.dname"학과이름" , d2.dname"소속"
FROM department d1 join department d2 on (d1.part =d2.deptno)
WHERE  ( 
SELECT deptno 
FROM department 
WHERE dname like '공과대학' 
) IN (d1.part,d2.part)
