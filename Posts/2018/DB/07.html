SELECT name "이름", tel "연락처", CASE( SUBSTR (tel,1,INSTR(TEL, ')' )-1 )) 
                                                           WHEN '02' THEN '서울'  
                                                           WHEN '031' THEN '경기도'  
                                                           WHEN '051' THEN '부산' 
                                                           WHEN '052' THEN '울산' 
                                                           WHEN '053' THEN '대구' 
                                                           WHEN '055' THEN '경남'     
                                                           ELSE '기타' 
END  "지역"  
FROM student  
WHERE deptno1 =101

3.교수의 이름 월급 보너스 세금을 출력하시오.
4500이하15%
4500 ~8800 24%
8800초과 35% 

SELECT name "이름", pay "급여",NVL(bonus,0) "보너스", 
					CASE  
					WHEN (pay*12)+(NVL(bonus,0)*6) <= 4500 THEN (pay+NVL(bonus,0))*0.15
					WHEN (pay*12)+(NVL(bonus,0)*6) <= 8800 THEN (pay+NVL(bonus,0))*0.24
					ELSE (pay+NVL(bonus,0)*6)*0.35
				   	END  "세금"
FROM professor
