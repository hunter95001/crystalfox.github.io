CREATE OR REPLACE FUNCTION BMI
    (v_height student.height%TYPE,
    v_weight student.weight%TYPE)
    RETURN CHAR
IS
    v_bmi NUMBER;
    v_char CHAR(20);
BEGIN    
    v_bmi := v_weight/(power(v_height/100,2));
 
    v_char := CASE
        WHEN v_bmi < 18.50 THEN '저체중'
        WHEN v_bmi < 23 THEN '정상'
        WHEN v_bmi <25 THEN '과체중'
        ELSE '비만'
    END CASE;
    RETURN v_char ;
EXCEPTION
    WHEN  NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE(' 존재하지 않는 교수번호입니다.');
    WHEN  TOO_MANY_ROWS THEN
    DBMS_OUTPUT.PUT_LINE(' 값이 여러개 있습니다.');
END;
