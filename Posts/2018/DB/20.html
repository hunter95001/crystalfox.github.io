CREATE OR REPLACE PROCEDURE display_name
    (vno    NUMBER,
    vgrade    NUMBER)
IS
    vname    VARCHAR(15);
    vbirthday    DATE;
BEGIN
    SELECT name,birthday    INTO  vname,vbirthday
    FROM student
    WHERE studno =vno AND grade =vgrade;
    DBMS_OUTPUT.PUT_LINE(vno||'    '||vname ||'    '||vbirthday||'    '||vgrade);
END;
/

CREATE OR REPLACE PROCEDURE count_dept
    (vdname department.dname%TYPE)
IS
    vdeptno   department.deptno%TYPE;
    vcount    NUMBER;
BEGIN
        SELECT deptno    INTO vdeptno
        FROM department    
        WHERE dname = vdname;    
        SELECT count(*) INTO vcount
        FROM student
        WHERE deptno1 = vdeptno;
        DBMS_OUTPUT.PUT_LINE(vdname ||' 학생수는'|| vcount||'입니다');
END;
/
